1) function counter(){
    var counter = 0;
        
    function IncreaseCounter() {
        return counter += 1;
    };
        
    return IncreaseCounter;
    }
        
    var counter = counter();
    console.log(counter());
    console.log(counter());
    console.log(counter());
    console.log(counter());
    
OUTPUT:
1
2
3
4

Every time the function runs, the variable counter value gets updated, increasing by 1.
    


2)  let count = 0;
    (function () {
        if (count === 0) {
            let count = 1;
            console.log(count); // What is logged?
        }
        console.log(count); // What is logged?
    })();
    
OUTPUT:
1
0

Inside the if statement a new let variable of name "count" is initialized with value 1 and that is the output.
Then because of scoping, the 2nd console.log cannot access count=1. It accesses the global variable count initialized with zero.

    


3)  for (var i = 0; i < 3; i++)
    {
        setTimeout(function log() {
            console.log(i); // What is logged?
        }, 1000);
    }
    
OUTPUT:
3
3
3

After a delay of 1 second, the setTimeout() runs 3 times, each time printing the last updated value of i, that is 3.
